    \begin{algorithm*}
        \caption{Towers of Hanoi}
        \begin{algorithmic}[1]
        \REQUIRE A positive integer $n$
        \STATE Move $n-1$ from the left to the middle
        \STATE Move the last disk to the right
        \STATE Move $n-1$ from the middle to the right
        \end{algorithmic}
    \end{algorithm*}
